# Define a helper function to return cell values from the data frame
def _safe_get(row: pd.Series, key: str) -> float:
    try:
        return row[key]
    except KeyError:
        return np.nan


# Define a function to evaluate for 1.1 Migraine without aura
def evaluate_1_1(row: pd.Series) -> Tuple[bool, List[bool]]:
    cond1 = _safe_get(row, "I10") == 0
    val_I29 = _safe_get(row, "I29")
    cond2 = (not pd.isna(val_I29)) and (val_I29 != 0)

    feat1 = _safe_get(row, "I22") == 0
    feat2 = _safe_get(row, "I23") == 0
    feat3 = _safe_get(row, "I24") == 1

    val_I30 = _safe_get(row, "I30")
    feat4 = (not pd.isna(val_I30)) and (val_I30 != 0)

    feat_count = sum([bool(feat1), bool(feat2), bool(feat3), bool(feat4)])
    cond3 = feat_count >= 2

    cond4 = (
        (_safe_get(row, "I25") == 1 or _safe_get(row, "I26") == 1)
        or (_safe_get(row, "I27") == 1 and _safe_get(row, "I28") == 1)
    )

    conditions = [cond1, cond2, cond3, cond4]
    return all(conditions), conditions


# Define a function to evaluate for 1.2 Migraine with aura
def evaluate_1_2(row: pd.Series) -> Tuple[bool, List[bool]]:
    cond1 = _safe_get(row, "I20") == 1
    cond2 = any(_safe_get(row, key) == 1 for key in ["I11", "I12", "I13"])

    aura_keys = ["I14", "I15", "I16", "I17", "I18", "I19"]
    aura_count = sum((_safe_get(row, key) == 1) for key in aura_keys)
    cond3 = aura_count >= 3

    conditions = [cond1, cond2, cond3]
    return all(conditions), conditions


# Define a function to evaluate for 1.3 Chronic migraine
def evaluate_1_3(row: pd.Series) -> Tuple[bool, List[bool]]:
    q11, _ = evaluate_1_1(row)
    q12, _ = evaluate_1_2(row)
    cond1 = q11 or q12

    val_I21 = _safe_get(row, "I21")
    val_I31 = _safe_get(row, "I31")
    cond2 = (
        (not pd.isna(val_I21) and val_I21 != 0)
        or (not pd.isna(val_I31) and val_I31 != 0)
    )

    conditions = [cond1, cond2]
    return all(conditions), conditions


# Define a function to evaluate for 2.1 Infrequent tension-type headache
def evaluate_2_1(row: pd.Series) -> Tuple[bool, List[bool]]:
    cond1 = _safe_get(row, "I29") == 2
    cond2 = (_safe_get(row, "I10") == 0) or (_safe_get(row, "I34") == 2)

    feat1 = _safe_get(row, "I22") == 1
    feat2 = _safe_get(row, "I23") == 1
    feat3 = _safe_get(row, "I24") == 0

    val_I30 = _safe_get(row, "I30")
    feat4 = (not pd.isna(val_I30)) and (val_I30 != 2)

    feat_count = sum([bool(feat1), bool(feat2), bool(feat3), bool(feat4)])
    cond3 = feat_count >= 2

    cond4 = (_safe_get(row, "I25") == 0) and (_safe_get(row, "I26") == 0)
    cond5 = (_safe_get(row, "I27") == 0) or (_safe_get(row, "I28") == 0)
    cond6 = _safe_get(row, "I32") == 0

    conditions = [cond1, cond2, cond3, cond4, cond5, cond6]
    return all(conditions), conditions


# Define a function to evaluate for 2.2 Frequent tension-type headache
def evaluate_2_2(row: pd.Series) -> Tuple[bool, List[bool]]:

# Define a function to evaluate for 2.3 Chronic tension-type headache
def evaluate_2_3(row: pd.Series) -> Tuple[bool, List[bool]]:
    _, conds = evaluate_2_1(row)
    conds = conds.copy()
    conds[-1] = (_safe_get(row, "I32") == 2)
    return all(conds), conds


# Define a helper function to evaluate for cirterion C of TACs
def _cluster_autonomic_base(row: pd.Series) -> Tuple[bool, bool, bool]:
    autonomic = any(_safe_get(row, key) == 1 for key in ["I36", "I37", "I38", "I39", "I40"])
    cond4 = (_safe_get(row, "I41") == 1) or autonomic
    cond7 = _safe_get(row, "I22") == 0
    return cond4, autonomic, cond7


# Define a function to evaluate for 3.1 Cluster headache
def evaluate_3_1(row: pd.Series) -> Tuple[bool, List[bool]]:
    cond1 = (_safe_get(row, "I44") != 0)
    cond2 = (_safe_get(row, "I30") == 2)
    cond3 = (_safe_get(row, "I43") == 1)

    cond4, _, cond7 = _cluster_autonomic_base(row)

    val_I42 = _safe_get(row, "I42")
    cond5 = (not pd.isna(val_I42)) and (val_I42 != 2)
    cond6 = (_safe_get(row, "I34") != 0)

    conditions = [cond1, cond2, cond3, cond4, cond5, cond6, cond7]
    return all(conditions), conditions


# Define a function to evaluate for 3.2 Paroxysmal hemicrania
def evaluate_3_2(row: pd.Series) -> Tuple[bool, List[bool]]:
    cond1 = (_safe_get(row, "I44") == 2)
    cond2 = (_safe_get(row, "I30") == 2)
    cond3 = (_safe_get(row, "I43") == 1)

    cond4, _, cond7 = _cluster_autonomic_base(row)

    val_I42 = _safe_get(row, "I42")
    cond5 = (not pd.isna(val_I42)) and (val_I42 != 0)
    cond6 = (_safe_get(row, "I34") != 2)

    conditions = [cond1, cond2, cond3, cond4, cond5, cond6, cond7]
    return all(conditions), conditions


# Define a function to evaluate for 3.4 Hemicrania continua
def evaluate_3_4(row: pd.Series) -> Tuple[bool, List[bool]]:
    cond1 = (_safe_get(row, "I45") == 1)

    val_I30 = _safe_get(row, "I30")
    cond2 = (not pd.isna(val_I30)) and (val_I30 != 0)

    cond3 = (_safe_get(row, "I46") == 1)
    cond4, _, cond7 = _cluster_autonomic_base(row)

    conditions = [cond1, cond2, cond3, cond4, cond7]
    return all(conditions), conditions

# Define a function to evaluate for probable diagnoses
def evaluate_probable(base_eval: Callable[[pd.Series], Tuple[bool, List[bool]]], row: pd.Series) -> bool:
    qualifies, conds = base_eval(row)
    if qualifies:
        return False
    total = len(conds)
    satisfied = sum(bool(c) for c in conds)
    return satisfied >= (total - 1)

# Define a function to classify cases
def classify_case(row: pd.Series) -> str:

    # Evaluate for confirmed diagnoses
    if evaluate_1_3(row)[0]:
        return "1.3."
    if evaluate_1_2(row)[0]:
        return "1.2."
    if evaluate_1_1(row)[0]:
        return "1.1."
    if evaluate_2_1(row)[0]:
        return "2.1."
    if evaluate_2_2(row)[0]:
        return "2.2."
    if evaluate_2_3(row)[0]:
        return "2.3."
    if evaluate_3_1(row)[0]:
        return "3.1."
    if evaluate_3_2(row)[0]:
        return "3.2."
    if evaluate_3_4(row)[0]:
        return "3.4."

    # Evaluate for 1.5. Probable Migraine
    if (
        evaluate_probable(evaluate_1_1, row)
        or evaluate_probable(evaluate_1_2, row)
        or evaluate_probable(evaluate_1_3, row)
    ):
        return "1.5."

    # Evaluate for 2.4. Probable tension headache
    if (
        evaluate_probable(evaluate_2_1, row)
        or evaluate_probable(evaluate_2_2, row)
        or evaluate_probable(evaluate_2_3, row)
    ):
        return "2.4."

    # Evaluate for 3.5. Probable TACs
    if (
        evaluate_probable(evaluate_3_1, row)
        or evaluate_probable(evaluate_3_2, row)
        or evaluate_probable(evaluate_3_4, row)
    ):
        return "3.5."

    # Return unclassified otherwise
    return "Unclassified"


def classify_csv(input_csv: str) -> pd.DataFrame:
    
    # Load CSV
    df = pd.read_csv(input_csv)

    # Convert item columns to numeric except selected categorical features
    for col in df.columns:
        if isinstance(col, str) and col.startswith("I") and col not in {"I1", "I2", "I5", "I6", "I8"}:
            df[col] = pd.to_numeric(df[col], errors="coerce")

    # Classify rows
    df = df.copy()
    df["Automatic diagnosis"] = df.apply(classify_case, axis=1)

    return df

# Define a helper function to extract first and second levels from a code (e.g. '1.3.')
def parse_levels(code: str):
    match = re.match(r'(\d+)\.(\d+)', str(code))
    return (match.group(1), match.group(2)) if match else (None, None)

# Define a function to match first-level diagnosis
def first_level_accuracy(row):
    auto_first, _ = parse_levels(row["Automatic diagnosis"])
    spec_first, _ = parse_levels(row["I6"])
    return "Classified" if (auto_first is not None and auto_first == spec_first) else "Misclassified"

# Define a function to match second-level diagnosis (if first level is accurate)
def second_level_accuracy(row):
    auto_first, auto_second = parse_levels(row["Automatic diagnosis"])
    spec_first, spec_second = parse_levels(row["I6"])

    if auto_first is None or spec_first is None or auto_first != spec_first:
        return "Misclassified"

    return "Classified" if (auto_second is not None and auto_second == spec_second) else "Misclassified"

# Define a function to return overall classification accuracy
def overall_accuracy(row):
    if row["First level accuracy"] != "Classified":
        return "Misclassified"
    if row["Second level accuracy"] != "Classified":
        return "First level"
    return "Second level"

color_matrix = np.zeros(contingency.shape)

# Loop over matrix rows
for i, auto in enumerate(contingency.index):

    # loop over matrix columns
    for j, spec in enumerate(contingency.columns):

        # Skip empty cells
        if contingency.loc[auto, spec] == 0:
            color_matrix[i, j] = np.nan
            continue

        # Acquire the first and second level diagnoses for each cell
        a1, a2 = parse_levels(auto)
        s1, s2 = parse_levels(spec)

        # Handle non-standard labels
        if a1 is None or s1 is None:
            color_matrix[i, j] = 0  # red
            continue

        # Compare between specialist and automatic diagnosis to establish color
        if a1 == s1 and a2 == s2:
            color_matrix[i, j] = 2  # green
        elif a1 == s1:
            color_matrix[i, j] = 1  # orange
        else:
            color_matrix[i, j] = 0  # red

groups_age = [group["I7"].dropna().values

epsilon_sq_age = (kw_age.statistic - k_age + 1) / (n_age - k_age)

# Define a function to calculate Cramer's V effect size from contingency tables
def cramers_v(contingency_table):
    chi2 = chi2_contingency(contingency_table)[0]
    n = contingency_table.sum().sum()
    min_dim = min(contingency_table.shape) - 1
    return np.sqrt(chi2 / (n * min_dim))
